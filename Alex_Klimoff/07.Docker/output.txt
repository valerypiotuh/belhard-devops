### Содержимое dockerfile
FROM alpine:latest
LABEL maintainer="klimoff@tut.by"
RUN mkdir /app
WORKDIR /app
RUN mkdir /log
COPY entrypoint.sh /entrypoint.sh
ENTRYPOINT ["/entrypoint.sh"]

### Содержимое entrypoint.sh (файл находится в одной папке с докерфайлом)
#!/bin/sh
apk update && apk add curl wget git bash speedtest-cli
speedtest-cli > speddtest.log
